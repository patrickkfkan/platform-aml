setenv kernel_addr "0x82000000"
setenv initrd_addr "0x88080000"
setenv boot_start bootm ${kernel_addr} ${initrd_addr}
if fatload mmc 0 ${kernel_addr} uImage; then if fatload mmc 0 ${initrd_addr} uInitrd; then run boot_start; fi;fi;
if fatload usb 0 ${kernel_addr} uImage; then if fatload usb 0 ${initrd_addr} uInitrd; then run boot_start; fi;fi;
